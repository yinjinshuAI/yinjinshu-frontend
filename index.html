// ====================== 核心API配置（Vercel版，彻底删除腾讯云） ====================== const API_URL = "/api/chat"; const chatMessages = document.getElementById('chatMessages'); const userInput = document.getElementById('userInput'); const sendBtn = document.getElementById('sendBtn'); // 添加消息到聊天框 function addMessage(sender, content, isUser = false) { const messageDiv = document.createElement('div'); messageDiv.className = `message ${isUser ? 'user' : ''}`; const avatarDiv = document.createElement('div'); avatarDiv.className = 'message-avatar'; avatarDiv.textContent = isUser ? '你' : 'AI'; const contentDiv = document.createElement('div'); contentDiv.className = 'message-content'; contentDiv.textContent = content; if (isUser) { messageDiv.appendChild(contentDiv); messageDiv.appendChild(avatarDiv); } else { messageDiv.appendChild(avatarDiv); messageDiv.appendChild(contentDiv); } chatMessages.appendChild(messageDiv); chatMessages.scrollTop = chatMessages.scrollHeight; } // 发送消息到AI async function sendMessage() { const msg = userInput.value.trim(); if (!msg) return; // 添加用户消息 addMessage('user', msg, true); userInput.value = ''; try { // 调用Vercel后端API（彻底告别腾讯云） const response = await fetch(API_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message: msg }) }); if (!response.ok) { throw new Error(`请求失败: ${response.status}`); } const data = await response.json(); // 添加AI回复 const aiReply = data.choices?.[0]?.message?.content || '收到你的消息，后端已连通！'; addMessage('ai', aiReply); } catch (error) { addMessage('ai', `发送失败: ${error.message}`); console.error('请求错误:', error); } } // 绑定发送事件 sendBtn.addEventListener('click', sendMessage); userInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') { sendMessage(); } });
